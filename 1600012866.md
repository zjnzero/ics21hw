int float_f2i(float_bits f){
    int sign = (f >> 31) & 1;
    int exp = (f >> 23) & 0xff - 127; // 计算2的幂指数部分
  	int frac = f & 0x7fffff;
    if (exp >= 32 || (exp == 31 && !frac)){ // INF的情况需要向0舍入
        return 0x80000000;
    }else if (exp < 0){
        return 0;
    }
  	frac += 0x800000; // 将尾数加上隐藏的1
    int frac = f & 0x7fffff + 0x800000; // 将尾数加上隐藏的1
    if (exp >= 23){
        if (sign)
            return -(frac << (exp - 23));
        else
            return (frac << (exp - 23));
    }else{
        if (sign)
            return -(frac >> (23 - exp)); //直接右移可以达到向0取整的目的
        else
            return (frac >> (exp - 23));
    }
}
